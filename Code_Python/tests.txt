import unittest
from pokemon import Pokemon
from pokeball import Pokeball

class TestPokemonProject(unittest.TestCase):
    
    def setUp(self):
        """Pr√©pare les objets avant chaque test (comme @BeforeEach)"""
        self.funecire = Pokemon("Feu", "Spectre")
        self.super_ball = Pokeball(600, self.funecire)
        self.funecire.pokeball = self.super_ball

    def test_type_display(self):
        self.assertEqual(self.super_ball.afficher_type_pokemon(), "De types Feu et Spectre.")

if __name__ == '__main__':
    unittest.main()
